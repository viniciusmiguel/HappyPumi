```csharp
public sealed record WebhookScope(string Kind, string Id); // Kind: "stack"|"org"|"environment"; Id: scope key e.g. "org/proj/stack"
public sealed class StoredWebhookDelivery {
    public required string Id { get; init; }
    public required WebhookScope Scope { get; init; }
    public required string WebhookName { get; init; }
    public required string Event { get; init; }
    public string RequestBody { get; set; } = "";
    public int ResponseStatus { get; set; }     // 0 = not sent / blocked
    public string? ResponseBody { get; set; }
    public long DurationMs { get; set; }
    public DateTime Timestamp { get; set; } = DateTime.UtcNow;
}
public interface IWebhookDeliveryStore {
    StoredWebhookDelivery Append(StoredWebhookDelivery delivery);
    IReadOnlyList<StoredWebhookDelivery> List(WebhookScope scope, string webhookName);
    StoredWebhookDelivery? LatestByEvent(WebhookScope scope, string webhookName, string @event);
}
```

**Produces:**
```csharp
public interface IWebhookDispatcher {
    // Fires to each active webhook whose filters match; renders+signs+POSTs; records a delivery; returns them.
    Task<IReadOnlyList<StoredWebhookDelivery>> FireAsync(
        WebhookScope scope, IEnumerable<WebhookResponse> webhooks, string @event, object payload, CancellationToken ct);
    // Single-webhook send for ping/redeliver (raw body already chosen). Records + returns the delivery.
    Task<StoredWebhookDelivery> SendAsync(
        WebhookScope scope, WebhookResponse webhook, string @event, string body, CancellationToken ct);
}
```
- Dispatcher: filter match = webhook.Filters empty (all) OR contains `@event`; render body via the formatter for `webhook.Format` (default raw = JSON of payload); `X-Pulumi-Signature: sha256=<hmac>` when `secret` present (see `WebhookSignature.Sign(body, secret)`); SSRF guard checks the URL host against `Webhooks:BlockedHosts` config (CSV) → blocked ⇒ record delivery with ResponseStatus 0 + note, skip POST; POST via typed `HttpClient`, capture status/body/elapsed; append to `IWebhookDeliveryStore`; never throw (catch network errors → ResponseStatus 0, recorded).
