# Pulumi Automation API Support — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Prove and harden HappyPumi as a backend for the Pulumi Automation API by adding a Go-auto-SDK integration-test layer that exercises every backend-touching operation, filling the gaps it surfaces (notably `DeleteStackTag` and the git-source remote-workspace path).

**Architecture:** A fourth integration-test layer (`HappyPumi.AutomationApi.IntegrationTests`) drives the Go auto SDK (which shells out to the locally-built `pulumi` binary) against a live `HappyPumiServer`, mirroring the existing CLI wire-compat layer. Endpoint gaps are fixed TDD-style in `HappyPumi.Api`. Remote workspaces add a git-source mode to `CreateApiDeploymentHandlerV2` + the runner job builder, verified by a Docker-backed test reusing the existing agent compose topology.

**Tech Stack:** .NET 10 / C# / FastEndpoints / EF Core + Postgres (Testcontainers); Go auto SDK (`github.com/pulumi/pulumi/sdk/v3/go/auto`) resolved against the local `../pulumi` checkout; xUnit; Docker (remote phase only).

## Global Constraints

- Clean-room: derived only from the public OpenAPI spec + Apache-2.0 CLI (ADR-0008). No proprietary source.
- Functions 4–20 lines; files < 500 lines; explicit types, no `any`/untyped; early returns, max 2 indent levels.
- TDD: every new function gets a test; bug fixes get a regression test. Mock external I/O with named fakes.
- Exception messages include the offending value and expected shape.
- Generated `// <auto-generated />` endpoint files: edit only the `HandleAsync` body (the team already does this; the generator-overwrite caveat in HappyPumi/CLAUDE.md is mitigated by committing before any regen). Never hand-edit generated *contract* files.
- Coverage > 80%, duplication < 3%, lint clean — checked locally before the PR.
- Commits: granular, WHAT/WHY focus, `Co-Authored-By: Claude Opus 4.8 <noreply@anthropic.com>` trailer. Do not squash.
- The Go module must build offline by resolving `sdk/v3/go/auto` to the local `../../../pulumi/sdk` via a `replace` directive (same pattern `make pulumi` relies on).

- [ ] **Step 5: Write `GoTestRunner.cs`** — runs `go test -json` and surfaces subtests

```csharp
using System.Diagnostics;
using System.Text;
using System.Text.Json;

namespace HappyPumi.AutomationApi.IntegrationTests;

/// <summary>
/// Runs the Go auto-SDK test module (autoapi/) as a child `go test -json` process against a live
/// HappyPumi. The Go SDK shells out to the pulumi binary, so .tools/bin is prepended to PATH and the
/// dev cert is trusted via SSL_CERT_DIR (ADR-0007), exactly as the CLI wire-compat layer does.
/// </summary>
public static class GoTestRunner
{
    public sealed record Result(int ExitCode, string Output);

    public static async Task<Result> Run(string backendUrl, string token, string? certDir, string runFilter, CancellationToken ct)
    {
        var moduleDir = Path.Combine(RepoPaths.RepoRoot, "HappyPumi.AutomationApi.IntegrationTests", "autoapi");
        var psi = new ProcessStartInfo("go")
        {
            RedirectStandardOutput = true,
            RedirectStandardError = true,
            UseShellExecute = false,
            WorkingDirectory = moduleDir,
        };
        psi.ArgumentList.Add("test");
        psi.ArgumentList.Add("-json");
        psi.ArgumentList.Add("-count=1");
        if (!string.IsNullOrEmpty(runFilter)) { psi.ArgumentList.Add("-run"); psi.ArgumentList.Add(runFilter); }
        psi.ArgumentList.Add("./...");

        psi.Environment["PULUMI_BACKEND_URL"] = backendUrl;
        psi.Environment["PULUMI_ACCESS_TOKEN"] = token;
        psi.Environment["PULUMI_SKIP_UPDATE_CHECK"] = "true";
        var binDir = Path.GetDirectoryName(RepoPaths.PulumiBinary ?? "");
        if (!string.IsNullOrEmpty(binDir))
            psi.Environment["PATH"] = binDir + Path.PathSeparator + (Environment.GetEnvironmentVariable("PATH") ?? "");
        if (!string.IsNullOrEmpty(certDir))
            psi.Environment["SSL_CERT_DIR"] = certDir;

        using var process = new Process { StartInfo = psi };
        var raw = new StringBuilder();
        process.OutputDataReceived += (_, e) => { if (e.Data is not null) lock (raw) raw.AppendLine(e.Data); };
        process.ErrorDataReceived += (_, e) => { if (e.Data is not null) lock (raw) raw.AppendLine(e.Data); };
        process.Start();
        process.BeginOutputReadLine();
        process.BeginErrorReadLine();
        await process.WaitForExitAsync(ct);

        return new Result(process.ExitCode, Summarize(raw.ToString()));
    }

    /// <summary>Flattens `go test -json` action events into readable PASS/FAIL lines per test.</summary>
    private static string Summarize(string jsonLines)
    {
        var sb = new StringBuilder();
        foreach (var line in jsonLines.Split('\n', StringSplitOptions.RemoveEmptyEntries))
        {
            if (!line.StartsWith('{')) { sb.AppendLine(line); continue; }
            try
            {
                using var doc = JsonDocument.Parse(line);
                var root = doc.RootElement;
                var action = root.TryGetProperty("Action", out var a) ? a.GetString() : null;
                if (action is "pass" or "fail" or "skip" && root.TryGetProperty("Test", out var t))
                    sb.AppendLine($"{action!.ToUpperInvariant()} {t.GetString()}");
                else if (action == "output" && root.TryGetProperty("Output", out var o))
                    sb.Append(o.GetString());
            }
            catch (JsonException) { sb.AppendLine(line); }
        }
        return sb.ToString();
    }
}
```

    var row = Row(c);

    return (ToStored(row, LoadHistory(c)), true);

func TestInlineUpPreviewRefreshDestroy(t *testing.T) {
	ctx := context.Background()
	proj := "hp-auto-life"
	name := fmt.Sprintf("organization/%s/s%d", proj, 1)
	prog := func(c *pulumi.Context) error { c.Export("ok", pulumi.String("yes")); return nil }
	s, err := auto.UpsertStackInlineSource(ctx, name, proj, prog)
	if err != nil { t.Fatalf("UpsertStackInlineSource: %v", err) }
	defer func() {
		_, _ = s.Destroy(ctx, optdestroy.ProgressStreams(&bytes.Buffer{}))
		_ = s.Workspace().RemoveStack(ctx, name)
	}()

	if _, err := s.Preview(ctx, optpreview.ProgressStreams(&bytes.Buffer{})); err != nil {
		t.Fatalf("Preview: %v", err)
	}
	up, err := s.Up(ctx, optup.ProgressStreams(&bytes.Buffer{}))
	if err != nil { t.Fatalf("Up: %v", err) }
	if up.Outputs["ok"].Value != "yes" { t.Fatalf("output mismatch: %v", up.Outputs) }

	if _, err := s.Refresh(ctx, optrefresh.ProgressStreams(&bytes.Buffer{})); err != nil {
		t.Fatalf("Refresh: %v", err)
	}
	hist, err := s.History(ctx, 10, 1)
	if err != nil { t.Fatalf("History: %v", err) }
	if len(hist) == 0 { t.Fatal("History empty after up") }

	outs, err := s.Outputs(ctx)
	if err != nil { t.Fatalf("Outputs: %v", err) }
	if outs["ok"].Value != "yes" { t.Fatalf("Outputs mismatch: %v", outs) }
}

- [ ] **Step 4: Add `GitDeployScript`** (mirror `DeployScript`'s hardening)

```csharp
/// <summary>Clones the deployment's git source, enters the project subdir, selects the stack, and runs the
/// requested Pulumi operation against HappyPumi — one step (see BuildJob for why). Repo URL is validated as
/// an http(s) git URL; branch/dir/coords go through the OWASP-A03 identifier allow-list (see DeployScript).</summary>
private static string GitDeployScript(DeploymentRow row, string backendUrl)
{
    var url = RequireGitUrl(row.GitRepoUrl!);
    var org = RequireSafeIdentifier(row.Org, nameof(row.Org));
    var project = RequireSafeIdentifier(row.Project, nameof(row.Project));
    var stack = RequireSafeIdentifier(row.Stack, nameof(row.Stack));
    var stackRef = $"{org}/{project}/{stack}";
    var op = row.Operation is "preview" or "refresh" or "destroy" ? row.Operation : "up";
    var apply = op == "preview" ? "pulumi preview" : $"pulumi {op} --yes --skip-preview";
    var branchArg = string.IsNullOrWhiteSpace(row.GitBranch) ? "" : $" -b '{RequireSafeIdentifier(row.GitBranch!, nameof(row.GitBranch))}'";
    var cd = string.IsNullOrWhiteSpace(row.GitRepoDir) ? "" : $"cd '{RequireSafeIdentifier(row.GitRepoDir!, nameof(row.GitRepoDir))}'\n";
    return string.Join('\n',
        "set -euo pipefail",
        "rm -rf /tmp/hpwork && mkdir -p /tmp/hpwork && cd /tmp/hpwork",
        $"git clone{branchArg} '{url}' repo",
        "cd repo",
        cd.TrimEnd('\n'),
        $"pulumi stack select --create '{stackRef}'",
        apply);
}

// http(s) git URL allow-list (OWASP A03): scheme + host/path chars only; excludes shell metacharacters,
// quotes, and whitespace so the value cannot break out of the surrounding single quotes.
private static readonly Regex GitUrl = new(@"^https?://[A-Za-z0-9._~:/?#@!$&()*+,;=%-]+$", RegexOptions.CultureInvariant);

private static string RequireGitUrl(string value)
{
    if (string.IsNullOrEmpty(value) || !GitUrl.IsMatch(value))
        throw new ArgumentException($"git repoUrl '{value}' is not a safe http(s) URL.", nameof(value));
    return value;
}
```
